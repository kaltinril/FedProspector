**`parse_date()` — test all format branches:**
```python
@pytest.mark.parametrize("value,expected", [
    ("2024-01-15", "2024-01-15"),           # ISO
    ("20240115", "2024-01-15"),             # compact YYYYMMDD
    ("01/15/2024", "2024-01-15"),           # US MM/DD/YYYY
    ("01-15-2024", "2024-01-15"),           # dash-US
    ("2024-01-15T14:30:00", "2024-01-15"), # ISO with time
    ("2024-01-15T00:00:00-05:00", "2024-01-15"),  # ISO with timezone
    ("", None),
    (None, None),
    ("not-a-date", None),
    ("  2024-01-15  ", "2024-01-15"),       # whitespace
    (20240115, "2024-01-15"),              # integer input
])
def test_parse_date(value, expected):
    assert parse_date(value) == expected
```

**`parse_decimal()` — test edge cases:**
```python
@pytest.mark.parametrize("value,expected", [
    ("1500000.00", "1500000.00"),
    ("1,500,000.00", "1500000.00"),   # comma stripping
    (1500000, "1500000"),
    (None, None),
    ("", None),                        # empty string → None (InvalidOperation)
    ("not-a-number", None),
])
def test_parse_decimal(value, expected):
    result = parse_decimal(value)
    assert result == expected
```

### Gap 5 — HIGH: `get_cursor()` Context Manager Not Tested

**File:** `fed_prospector/tests/test_db_pool.py`

Add tests to the existing file:
```python
def test_get_cursor_yields_cursor(mock_db_connection):
    with get_cursor() as cursor:
        assert cursor is not None

def test_get_cursor_closes_on_exit(mock_db_connection):
    mock_conn = mock_db_connection.return_value
    mock_cursor = mock_conn.cursor.return_value
    with get_cursor():
        pass
    mock_cursor.close.assert_called_once()
    mock_conn.close.assert_called_once()

def test_get_cursor_closes_on_exception(mock_db_connection):
    mock_conn = mock_db_connection.return_value
    mock_cursor = mock_conn.cursor.return_value
    with pytest.raises(ValueError):
        with get_cursor():
            raise ValueError("test error")
    mock_cursor.close.assert_called_once()
    mock_conn.close.assert_called_once()

def test_get_cursor_dictionary_param(mock_db_connection):
    mock_conn = mock_db_connection.return_value
    with get_cursor(dictionary=True):
        pass
    mock_conn.cursor.assert_called_with(dictionary=True)
```
